je_fatamid_india_gdp = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		capital={
			OR={
				region = sr:region_south_india
				region = sr:region_north_india
			}
		}	
		hidden_trigger={
			THIS = c:W57
		}
	}

	possible = {
		hidden_trigger={
			THIS = c:W57
		}
		gdp > 5000000
		custom_tooltip={
			text = je_fatamid_india_conquest_tt_5
			capital={
				OR={
					region = sr:region_south_india
					region = sr:region_north_india
				}
			}
		}
	}


	complete = {
		calc_true_if = {
			amount >= 3
			gdp > 10000000
			gdp > 20000000
			custom_tooltip={
				text = je_fatimid_indian_gdp_tt_1
				NOR = {
					sr:region_south_india = {
						any_scope_state = {
							is_capital = yes
							owner={
								gdp > root.gdp
							}
						}
					}
					sr:region_north_india = {
						any_scope_state = {
							is_capital = yes
							owner={
								gdp > root.gdp
							}
						}
					}
				}
			}
			custom_tooltip={
				text = je_fatimid_indian_gdp_tt_2
				OR={
					is_subject = no 
					NOT = {
						any_in_hierarchy = {
							NOR={
								THIS = ROOT
								THIS = root.overlord
							}
							gdp > root.gdp
						}
					}
				}
			}
			average_sol >= 15
		}
	}	

	invalid = {
		NOT={
			custom_tooltip={
				text = je_fatamid_india_conquest_tt_4
				capital={
					OR={
						region = sr:region_south_india
						region = sr:region_north_india
					}
				}
			}
		}
	}
	
	on_complete = {
		add_modifier = {
			name = ceylon_gdp_modifier
			months = normal_modifier_time
		}
	}


	weight = 100

	should_be_pinned_by_default = yes
}
