namespace = fat_vas

# Fatamid Vassal absorbs poorer neighbour
fat_vas.1 = {
	type = country_event
	placement = ROOT

	title = fat_vas.1.t
	desc = fat_vas.1.d
	flavor = fat_vas.1.f

	event_image = {
		video = "ep1_transfer_of_authority"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		top_overlord = {
			random_subject_or_below = {
				limit = {
					OR={
						any_neighbouring_state = {
							owner={
								THIS = ROOT
							}
						}
						OR = {
							any_scope_state = {
								any_sea_node_adjacent_state = {
									owner = {
										OR={
											owns_entire_state_region = STATE_NORTH_MADAGASCAR
											owns_entire_state_region = STATE_SOUTH_MADAGASCAR
										}
									}
									ROOT={
										NOR={
											owns_entire_state_region = STATE_NORTH_MADAGASCAR
											owns_entire_state_region = STATE_SOUTH_MADAGASCAR
										}
									}
								}
							}
							any_scope_state = {
								any_sea_node_adjacent_state = {
									owner = {
										NOR={
											owns_entire_state_region = STATE_NORTH_MADAGASCAR
											owns_entire_state_region = STATE_SOUTH_MADAGASCAR
										}
									}
									ROOT={
										OR={
											owns_entire_state_region = STATE_NORTH_MADAGASCAR
											owns_entire_state_region = STATE_SOUTH_MADAGASCAR
										}
									}
								}
							}
						}						
					}
					is_at_war = no				
					any_primary_culture = {
						has_discrimination_trait = african_heritage 
					}
					is_ai = yes
					this.gdp < root.gdp					
					NOT={
						any_primary_culture={
							culture_is_discriminated_in = ROOT
						}
					}
					OR = {
						owns_entire_state_region = STATE_UGANDA
						owns_entire_state_region = STATE_KENYA
						owns_entire_state_region = STATE_RIFT_VALLEY
						owns_entire_state_region = STATE_TANGANYIKA
						owns_entire_state_region = STATE_LINDI
						owns_entire_state_region = STATE_ZANZIBAR
						owns_entire_state_region = STATE_MOCAMBIQUE
						owns_entire_state_region = STATE_KAZEMBE
						owns_entire_state_region = STATE_ZAMBEZI
						owns_entire_state_region = STATE_ZAMBIA
						owns_entire_state_region = STATE_ZAMBEZIA
						owns_entire_state_region = STATE_LOURENCO_MARQUES
						owns_entire_state_region = STATE_NORTH_MADAGASCAR
						owns_entire_state_region = STATE_SOUTH_MADAGASCAR
					}
					is_subject = yes
					NOT = { THIS = ROOT }
					top_overlord = {
						THIS = root.top_overlord
					}
				}
				save_scope_as = alwatan_annex_scope
			}
		}

		top_overlord = {
			save_scope_as = alwatan_overlord_country
			ruler = {
				save_scope_as = alwatan_overlord_ruler
			}
		}
	}

	option = {
		name = fat_vas.1.a
		default_option = yes
		hidden_effect = {
			every_state = {
				limit = {
					has_claim_by = scope:alwatan_annex_scope
				}
				state_region = {
					remove_claim = scope:alwatan_annex_scope
					add_claim = ROOT
				}
			}
		}
		annex_with_incorporation = scope:alwatan_annex_scope
	}
}

# Fatamid combines two vassals
fat_vas.2 = {
	type = country_event
	placement = ROOT

	title = can_aus.2.t
	desc = can_aus.2.d
	flavor = can_aus.1.f

	event_image = {
		video = "ep1_power_blocs"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		random_subject_or_below = {
			limit = {
				any_primary_culture = { is_target_in_global_variable_list = { name = global_canadian_cultures target = this } }		##Converter
				capital = { zzz_converter_state_is_in_canada = yes }									##Converter
				relations:root >= relations_threshold:amicable
				is_ai = yes
			}
			save_scope_as = canada_scope_1
		}
		random_subject_or_below = {
			limit = {
				any_primary_culture = { is_target_in_global_variable_list = { name = global_canadian_cultures target = this } }		##Converter
				capital = { zzz_converter_state_is_in_canada = yes }									##Converter
				relations:root >= relations_threshold:amicable
				is_ai = yes
				NOT = {
					scope:canada_scope_1 = THIS
				}
			}
			save_scope_as = canada_scope_2
		}
	}

	option = {
		name = can_aus.2.a
		default_option = yes
		if = {
			limit = {
				scope:canada_scope_1.gdp > scope:canada_scope_2.gdp
			}
			hidden_effect = {
				every_state = {
					limit = {
						has_claim_by = scope:canada_scope_2
					}
					state_region = {
						remove_claim = scope:canada_scope_2
						add_claim = scope:canada_scope_1
					}
				}
			}
			scope:canada_scope_1 = {
				annex_with_incorporation = scope:canada_scope_2
			}
		}
		else = {
			hidden_effect = {
				every_state = {
					limit = {
						has_claim_by = scope:canada_scope_1
					}
					state_region = {
						remove_claim = scope:canada_scope_1
						add_claim = scope:canada_scope_2
					}
				}
			}
			scope:canada_scope_2 = {
				annex_with_incorporation = scope:canada_scope_1
			}
		}
	}
}


# Fatamid vassal completion for vassal
fat_vas.3 = {
	type = country_event
	placement = ROOT

	title = fat_vas.5.t
	desc = fat_vas.5.d
	flavor = fat_vas.5.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		top_overlord = {
			save_scope_as = alwatan_overlord_country
		}
	}

	option = {
		name = fat_vas.5.a
		add_modifier = {
			name = alwatan_unified
			months = normal_modifier_time
		}
		if = {
			limit = {
				OR={
					NOT = { is_subject_type = subject_type_dominion }
					NOT = { is_subject_type = subject_type_protectorate }
				}
			}
			change_subject_type = subject_type_dominion
		}
		##Converter
		if = {
			limit = {
				top_overlord = {
					any_scope_state = { trigger_state_is_in_alwatan = yes }
				}
			}
			top_overlord = {
				every_scope_state = {
					limit = { trigger_state_is_in_alwatan = yes }
					set_state_owner = root
				}
			}
		}
		change_tag = AWT
		##Converter
	}
}

# Fatamid vassal completion for overlord
fat_vas.4 = {
	type = country_event
	placement = ROOT

	title = can_aus.5.t
	desc = can_aus.6.d
	flavor = can_aus.5.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		random_subject_or_below = {
			limit = {
				any_primary_culture = { is_target_in_global_variable_list = { name = global_canadian_cultures target = this } }		##Converter
				capital = { zzz_converter_state_is_in_canada = yes }										##Converter
				any_scope_state = {
					OR = {
						state_region = s:STATE_ONTARIO
						state_region = s:STATE_NEW_BRUNSWICK
						state_region = s:STATE_QUEBEC
						state_region = s:STATE_NEWFOUNDLAND
						state_region = s:STATE_ALBERTA
						state_region = s:STATE_BRITISH_COLUMBIA
						state_region = s:STATE_SASKATCHEWAN
						state_region = s:STATE_MANITOBA
						state_region = s:STATE_YUKON_TERRITORY
						state_region = s:STATE_NORTHWEST_TERRITORIES
						state_region = s:STATE_NUNAVUT
					}
					count >= 9
				}
			}
			save_scope_as = canada_country
		}
		##Converter
		scope:canada_country = {
			if = {
				limit = {
					has_game_rule = converter_native_incorporation_access_addition
					any_primary_culture = {
						has_discrimination_trait = indigenous_american_heritage
						is_target_in_global_variable_list = { name = global_canadian_cultures target = this }
						NOT = { is_target_in_global_variable_list = { name = global_colonial_primary_cultures target = this } }
					}
				}
				every_primary_culture = {
					limit = {
						has_discrimination_trait = indigenous_american_heritage
						is_target_in_global_variable_list = { name = global_canadian_cultures target = this }
						NOT = { is_target_in_global_variable_list = { name = global_colonial_primary_cultures target = this } }
					}
					add_to_global_variable_list = {
						name = global_colonial_primary_cultures
						target = this
					}
				}
			}
		}
	}

	option = {
		name = can_aus.5.a
		add_modifier = {
			name = gbr_unified_canada
			months = normal_modifier_time
		}
		scope:canada_country = { change_tag = CAN }		##Converter
	}

	after = {		##Converter
		c:CAN = {
			every_primary_culture = {
				c:CAN = { remove_primary_culture = prev }
			}
			every_scope_state = {
				limit = { zzz_converter_state_is_in_canada = yes }
				every_scope_pop = {
					limit = {
						culture = {
							is_target_in_global_variable_list = { name = global_colonial_primary_cultures target = this }
							is_target_in_global_variable_list = { name = global_canadian_cultures target = this }
						}
					}
					culture = {
						c:CAN = { add_primary_culture = prev }
					}
				}
			}
		}
	}
}
